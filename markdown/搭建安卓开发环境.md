# 1 搭建环境

> ADT 

#### 1.1 下载 ADT Bundle 开发工具集合

 	1. 下载 adt-bundle-windows-x86_64_20140101 ，然后解压到本地磁盘上
 	2.  打开 SDKManager.exe ，下载相应的安装包，不需要全部下载
 	3.  将 platform-tools 目录下的 adb.exe ，添加到环境变量中，方便之后调试

####  1.2 实现 HelloWorld 

1. 新建一个 安卓模拟器，注意打开后不要关闭了，方便后面项目使用，DDMS
2. 新建一个 android application project 项目
3. 点击 Run 运行该HelloWorld 项目，即可在安卓模拟器看到 HelloWorld 项目的app
4. 打包 apk导出到本地磁盘

## 2.1 相对布局



```xml
<!-- RelationLayout 相对布局  -->
<RelativeLayout xmlns:android="http://schemas.android.com/apk/res/android"
    xmlns:tools="http://schemas.android.com/tools"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    android:gravity="top"
    android:paddingBottom="@dimen/activity_vertical_margin"
    android:paddingLeft="@dimen/activity_horizontal_margin"
    android:paddingRight="@dimen/activity_horizontal_margin"
    android:paddingTop="@dimen/activity_vertical_margin"
    tools:context=".HelloWorldActivity" >

    <!--
    
    	 文本控件： 设置 RelationLayout 宽         高
    	 match_parent 使视图扩展到父元素大小  
  
      -->
    <TextView
        android:id="@+id/textView1"
        android:layout_width="match_parent" 
        android:layout_height="match_parent"
        android:text="@string/hello_world" />
	
    
    <!--
    	 centerHorizeontal 居中
    	 wrap_content   适配屏幕
    	
     -->
    <Button 
        android:id="@+id/button1"
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:layout_alignParentTop="true"
        android:layout_centerHorizontal="true"
        android:layout_marginTop="120dp"
        android:text="@string/login"  />
    
    <!-- 注意 string 中 s 小写 -->
    <Button 
        android:id="@+id/button2"
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:layout_alignBottom="@+id/button1"
    	android:layout_marginBottom="50dp" 
        android:layout_toRightOf="@+id/button1"
        android:text="@string/regsiter"  />
    
</RelativeLayout>
```

## 2.2 流式布局

> 应用场景：用户登录

```xml
<?xml version="1.0" encoding="utf-8"?>
<LinearLayout xmlns:android="http://schemas.android.com/apk/res/android"
    android:layout_width="match_parent"
    android:layout_height="match_parent" 
    android:orientation="horizontal" >
	
    <!-- 
    		LinearLayout 线性布局
    		android:orientation="vertical" 表示垂直  vertical  默认
    		android:orientation="horizontal" 表示水平 horizontal
    		
     -->
    
    <Button
        	android:id="@+id/button1" 
        	android:layout_width="wrap_content"
        	android:layout_height="wrap_content" 
        	android:text="@string/button1"    />
    <Button
        	android:id="@+id/button2" 
        	android:layout_width="wrap_content"
        	android:layout_height="wrap_content" 
        	android:text="@string/button2"    />
    
    <Button
        	android:id="@+id/button3" 
        	android:layout_width="wrap_content"
        	android:layout_height="wrap_content" 
        	android:text="@string/button3"    />
</LinearLayout>
```

## 2.4 网格布局

> 从安卓 4.0 以后开始支持的
>
> 所需的  SDK 版本最低位 14
>
> 应用场景：布局

```xml
<?xml version="1.0" encoding="utf-8"?>
<GridLayout xmlns:android="http://schemas.android.com/apk/res/android"
    android:layout_width="match_parent"
    android:layout_height="match_parent" 
    android:layout_gravity="center"
    android:columnCount="4"
    android:orientation="horizontal">
    
	<Button 
	    android:layout_column="3"
	    android:text="@string/div"
	 />
	<Button android:text="@string/s1"/>
    <Button android:text="@string/s2"/>
    <Button android:text="@string/s3"/>
    <Button android:text="@string/plus"/>
    
    <Button android:text="@string/s4"/>
    <Button android:text="@string/s5"/>
    <Button android:text="@string/s6"/>
    <Button android:text="@string/sub"/>
    
     <Button android:text="@string/s7"/>
    <Button android:text="@string/s8"/>
    <Button android:text="@string/s9"/>
    <Button 
        android:layout_gravity="fill"
        android:layout_rowSpan="3"
        android:text="@string/add"/>
    
    <Button 
        android:layout_gravity="fill"
        android:layout_columnSpan="2"
        android:text="@string/zero"/>
    
    <Button android:text="@string/doublezero"/>
    
     <Button 
        android:layout_columnSpan="3"
```
